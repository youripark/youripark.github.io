Question 4

def shocker(n):
    charge = lambda f: lambda: f() * 2
    thunder = lambda: n
    def shock(power):
        nonlocal charge, thunder
        if power:
            thunder = charge(thunder)
        return thunder()
    return shock

pikachu = shocker(120)
damage = pikachu('PIKACHUUUUUUU!')
